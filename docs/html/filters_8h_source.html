198  void step(real_t control_vector, real_t measurement_vector)
199  {
200  //prediction
201  real_t predicted_state_estimate = (current_state_estimate) + (B * control_vector);
202  real_t predicted_prob_estimate = current_prob_estimate + Q;
203 
204  //observation
205  real_t innovation = measurement_vector - predicted_state_estimate;
206  real_t innovation_covariance = predicted_prob_estimate + R;
207 
208  //update
209  real_t kalman_gain = predicted_prob_estimate * innovation_covariance;
210  current_state_estimate = predicted_state_estimate + kalman_gain * innovation;
211  current_prob_estimate = (1 - kalman_gain) * predicted_prob_estimate;
212  }
213 
214 private:
215  real_t B, current_state_estimate, current_prob_estimate;
216  real_t Q, R;
